<?php
$f = 'data.json';
$d = json_decode(file_get_contents($f) ?: '[]', true);

if (isset($_POST['n'])) {
    if ($_POST['id']) {
        foreach($d as &$v) if($v['id'] == $_POST['id']) $v = $_POST;
    } else { $_POST['id'] = time(); $d[] = $_POST; }
    file_put_contents($f, json_encode($d)); header("Location: ?");
}

if (isset($_GET['del'])) {
    $d = array_values(array_filter($d, fn($v) => $v['id'] != $_GET['del']));
    file_put_contents($f, json_encode($d)); header("Location: ?");
}

$e = isset($_GET['edit']) ? current(array_filter($d, fn($v) => $v['id'] == $_GET['edit'])) :
['id' => '', 'n' => '', 'e' => '', 'p' => '', 'a' => '', 'g' => ''];
?>

<form method="post">
    <input type="hidden" name="id" value="<?=$e['id']?>">
    <input name="n" value="<?=$e['n']?>" placeholder="ناو" required>
    <input name="e" value="<?=$e['e']?>" placeholder="ئیمەیڵ" required>
    <input name="p" value="<?=$e['p']?>" placeholder="تەلەفۆن" required>
    <input name="a" value="<?=$e['a']?>" placeholder="شوێن" required>
    <select name="g">
        <option>نێر</option>
        <option <?=$e['g']=='مێ'?'selected':''?>>مێ</option>
    </select>
    <button>تۆمارکردن</button>
</form>

<table border="1" style="width:100%">
    <?php foreach($d as $v): ?>
    <tr>
        <td><?=$v['n']?></td>
        <td><?=$v['e']?></td>
        <td><?=$v['p']?></td>
        <td><?=$v['a']?></td>
        <td><?=$v['g']?></td>
        <td>
            <a href="?edit=<?=$v['id']?>">دەستکاری</a> | <a href="?del=<?=$v['id']?>">سڕینەوە</a>
        </td>
    </tr>
    <?php endforeach; ?>
</table>
